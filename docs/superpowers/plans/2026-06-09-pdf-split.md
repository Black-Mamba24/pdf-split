# pdf-split Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone cross-platform `pdf-split` CLI that splits one PDF into complete, ordered, continuous ranges by requested parts, strict maximum file size, or both.

**Architecture:** Keep all split decisions in a pure `planner` package whose plans are validated before use. Hide `pdfcpu` behind a small PDF engine interface, measure real candidate files for size-constrained planning, and generate final files into a same-filesystem staging directory before verified publication with rollback.

**Tech Stack:** Go 1.25, Cobra 1.10.2, pdfcpu 0.12.1, `golang.org/x/term`, Go standard testing/fuzzing, GitHub Actions, GoReleaser.

**Design reference:** `docs/superpowers/specs/2026-06-09-pdf-split-design.md`

Create `internal/cli/size.go`:

```go
package cli

import (
	"fmt"
	"math"
	"regexp"
	"strconv"
	"strings"
)

var sizePattern = regexp.MustCompile(`^([0-9]+(?:\.[0-9]+)?)(KB|MB|GB)$`)

func ParseSize(input string) (int64, error) {
	m := sizePattern.FindStringSubmatch(strings.ToUpper(strings.TrimSpace(input)))
	if m == nil {
		return 0, fmt.Errorf("invalid size %q: use a positive number followed by KB, MB, or GB", input)
	}
	value, err := strconv.ParseFloat(m[1], 64)
	if err != nil || value <= 0 {
		return 0, fmt.Errorf("invalid size %q", input)
	}
	multiplier := map[string]float64{"KB": 1 << 10, "MB": 1 << 20, "GB": 1 << 30}[m[2]]
	bytes := value * multiplier
	if math.IsInf(bytes, 0) || bytes > math.MaxInt64 {
		return 0, fmt.Errorf("size %q is too large", input)
	}
	return int64(math.Round(bytes)), nil
}
```

package domain

import "fmt"

type PageRange struct {
	Start int
	End   int
}

func (r PageRange) Pages() int { return r.End - r.Start + 1 }

type SplitPlan struct {
	Ranges []PageRange
}

func (p SplitPlan) Validate(totalPages int) error {
	if totalPages < 1 || len(p.Ranges) == 0 {
		return fmt.Errorf("plan must cover a non-empty PDF")
	}
	next := 1
	for i, r := range p.Ranges {
		if r.Start != next || r.End < r.Start || r.End > totalPages {
			return fmt.Errorf("invalid range %d: %d-%d, expected start %d", i+1, r.Start, r.End, next)
		}
		next = r.End + 1
	}
	if next != totalPages+1 {
		return fmt.Errorf("plan ends at page %d, want %d", next-1, totalPages)
	}
	return nil
}
```

func ByParts(totalPages, parts int) (domain.SplitPlan, error) {
	if totalPages < 1 || parts < 1 || parts > totalPages {
		return domain.SplitPlan{}, fmt.Errorf("parts must be between 1 and %d", totalPages)
	}
	base, remainder, next := totalPages/parts, totalPages%parts, 1
	ranges := make([]domain.PageRange, 0, parts)
	for i := 0; i < parts; i++ {
		count := base
		if i < remainder {
			count++
		}
		ranges = append(ranges, domain.PageRange{Start: next, End: next + count - 1})
		next += count
	}
	plan := domain.SplitPlan{Ranges: ranges}
	return plan, plan.Validate(totalPages)
}
```
